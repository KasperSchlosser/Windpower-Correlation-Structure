\section{Distributions}

\subsection{Beta Distribution}

\begin{equation}
    f(x; \alpha, \beta) = \frac{x^{\alpha-1}(1-x)^{\beta-1}}{B(\alpha, \beta)}
    \label{eq:beta_pdf}
\end{equation}

\begin{equation}
    B(\alpha, \beta) = \int_0^1 t^{\alpha-1} (1-t)^{\beta-1} \, dt
    \label{eq:beta_func}
\end{equation}

\begin{equation}
   f(x; \alpha, \beta, a, b) = \frac{(x-a)^{\alpha-1} (b-x)^{\beta-1}}{(b-a)^{\alpha+\beta-1} B(\alpha, \beta)}, \quad a \leq x \leq b
   \label{eq:beta_distribution}
\end{equation}
