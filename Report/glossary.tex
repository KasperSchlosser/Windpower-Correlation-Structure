% symbols and nomenclature
\newglossaryentry{range}
{
  name={\ensuremath{[a;b]}},
  description={Set of numbers between a and b, inclusive. $[a;b] = \{x \in \mathbb{R} | a \leq x \leq b\}$},
  type=symbols
}

\newglossaryentry{expected}
{
  name={\ensuremath{\mathbb{E}_P[x]}},
  description={The expected value of $x$ under probability measure $P$},
  type=symbols
}

\newglossaryentry{positive}
{
  name={\ensuremath{x^+}},
  description={The "positive" function: 
    \[
        x^+ = 
        \begin{cases} 
            x, & \text{if } x \geq 0 \\
            0, & \text{if } x < 0 
        \end{cases}
    \]},
  type=symbols
}



\newglossaryentry{Tranpose}
{
  name={\ensuremath{X^T}},
  description={The transpose of a matrix $X$
  \[
  X = \begin{bmatrix}
    a & b\\
    c  & d
  \end{bmatrix} \iff X^T = \begin{bmatrix}
    a & c\\
    b  & d
  \end{bmatrix}
  \]},
  type=symbols,
}


\newglossaryentry{indicator}
{
  name={$\mathbb{I}_{x < a}$},
  description={Indicator function with condition in subscript \[\mathbb{I}_{x<a} = \begin{cases}1 & x < a \\ 0 \end{cases}\]},
  type=symbols
}
